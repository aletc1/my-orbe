  kubectl -n {{ .Release.Namespace }} port-forward svc/{{ include "kyomiru.fullname" . }}-api 3000:3000
  curl http://localhost:3000/api/healthz

Run enrichment / backfill scripts against the live API pod:
  kubectl -n {{ .Release.Namespace }} exec deploy/{{ include "kyomiru.fullname" . }}-api -- npm run backfill:enrichment
  kubectl -n {{ .Release.Namespace }} exec deploy/{{ include "kyomiru.fullname" . }}-api -- npm run backfill:state
  kubectl -n {{ .Release.Namespace }} exec deploy/{{ include "kyomiru.fullname" . }}-api -- npm run backfill:translations
  kubectl -n {{ .Release.Namespace }} exec deploy/{{ include "kyomiru.fullname" . }}-api -- npm run backfill:reclassify
  kubectl -n {{ .Release.Namespace }} exec deploy/{{ include "kyomiru.fullname" . }}-api -- npm run recompute:airing
  kubectl -n {{ .Release.Namespace }} exec deploy/{{ include "kyomiru.fullname" . }}-api -- npm run merge:scan

Diagnose enrichment for a single show (dry-run; append -- --apply to persist):
  kubectl -n {{ .Release.Namespace }} exec deploy/{{ include "kyomiru.fullname" . }}-api -- npm run enrichment:debug -- <showId>

Manage invite-only access (requires DISABLE_AUTO_SIGNUP=true):
  kubectl -n {{ .Release.Namespace }} exec deploy/{{ include "kyomiru.fullname" . }}-api -- npm run approved:add -- you@example.com
  kubectl -n {{ .Release.Namespace }} exec deploy/{{ include "kyomiru.fullname" . }}-api -- npm run approved:list
  kubectl -n {{ .Release.Namespace }} exec deploy/{{ include "kyomiru.fullname" . }}-api -- npm run approved:remove -- you@example.com

Inspect BullMQ queue depth (waiting/active/failed counts + live events):
  kubectl -n {{ .Release.Namespace }} exec deploy/{{ include "kyomiru.fullname" . }}-api -- npm run queue:status
  kubectl -n {{ .Release.Namespace }} exec -it deploy/{{ include "kyomiru.fullname" . }}-api -- npm run queue:status -- --watch

Drain completed + failed BullMQ jobs (scope with --queue=<name> --state=<state> --waiting):
  kubectl -n {{ .Release.Namespace }} exec deploy/{{ include "kyomiru.fullname" . }}-api -- npm run queue:clean

Repair a show's catalog by replaying its watch_events (use --scan to list shows with lost progress, --scan --apply to bulk-rebuild):
  kubectl -n {{ .Release.Namespace }} exec deploy/{{ include "kyomiru.fullname" . }}-api -- npm run rebuild:show -- <showId>
  kubectl -n {{ .Release.Namespace }} exec deploy/{{ include "kyomiru.fullname" . }}-api -- npm run rebuild:show -- --scan

Prune phantom episodes left over from older catalog shapes (--scan to dry-run, --scan --apply to delete, or pass a <showId> to scope to one show):
  kubectl -n {{ .Release.Namespace }} exec deploy/{{ include "kyomiru.fullname" . }}-api -- npm run cleanup:phantom-eps -- --scan
  kubectl -n {{ .Release.Namespace }} exec deploy/{{ include "kyomiru.fullname" . }}-api -- npm run cleanup:phantom-eps -- --scan --apply
  kubectl -n {{ .Release.Namespace }} exec deploy/{{ include "kyomiru.fullname" . }}-api -- npm run cleanup:phantom-eps -- <showId> --apply
